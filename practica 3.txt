fun main() {
    // Lista de solo lectura
    val fruits = listOf("Uva", "Mango", "Sandía")
    println("Lista de frutas: $fruits")

    // Acceder a elementos
    println("Primera fruta: ${fruits[0]}")
    println("Total de frutas: ${fruits.size}")

    // Lista mutable
    val animals = mutableListOf("Tortuga", "Loro", "Panda", "Delfín")

    // Eliminar elementos
    animals.remove("Loro")
    println("Después de eliminar 'Loro': $animals")

    animals.removeAt(0)
    println("Después de eliminar el primer elemento: $animals")

    animals.removeAll(listOf("Panda"))
    println("Después de eliminar 'Panda': $animals")

    // Actualizar elementos
    val cities = mutableListOf("Madrid", "Roma", "Berlín")
    cities[1] = "Lisboa"
    println("\nCiudades actualizadas: $cities")

    for (i in cities.indices) {
        if (cities[i] == "Berlín") {
            cities[i] = "Atenas"
        }
    }
    println("Lista final de ciudades: $cities")
}
